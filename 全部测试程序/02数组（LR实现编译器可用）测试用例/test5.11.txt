//利用二维数组求每个学生的平均成绩和总平均
main()
{
    int a[5][3];//={{80,75,92},{61,65,71},{59,63,70},{85,87,90},{76,77,85}};
    int i,j,sum=0,average,v[3];
    a[0][0]=80;
    a[0][1]=75;
    a[0][2]=92;
    a[1][0]=61;
    a[1][1]=65;
    a[1][2]=71;
    a[2][0]=59;
    a[2][1]=63;
    a[2][2]=70;
    a[3][0]=85;
    a[3][1]=87;
    a[3][2]=90;
    a[4][0]=76;
    a[4][1]=77;
    a[4][2]=85;


    for(i=0;i<3;i=i+1)     //表示科目
    {
        for(j=0;j<5;j=j+1)  //表示学生
        {
            sum=sum+a[j][i];
        }
        v[i]=sum/5;
        sum=0;
    }
    average=(v[0]+v[1]+v[2])/3;
    write("math:    language:      C programming");
    write(v[0]);
    write(v[1]);
    write(v[2]);
    write("total average:");
    write(average);
}
