void quicksort(int first,int end)
{
    if(first > end)  //�ݹ��������
    {
        return;
    }
    int i = first, j = end, flag = arr[i], exchange = 0;

    while(i != j)      {
        while(i < j && arr[j] > flag)  
        {
            j=j-1;
        }
 
        while(i < j && arr[i] <= flag)
        {
            i=i+1;
        }
        if(j > i)
        {
            exchange = arr[i];
            arr[i] = arr[j];
            arr[j] = exchange;
        }
    }
    arr[first] = arr[i];
    arr[i] = flag;
    quicksort(first,i - 1);
    quicksort(i + 1,end);
}
