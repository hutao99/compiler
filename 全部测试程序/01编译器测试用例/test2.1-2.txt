main()
{
	int a=0;
	int max = 0;
	int min = 0;
	int sum = 0;
	int i = 0;
	while(i<5)
	{
		a = read();
		if(i==0){
			max = a;
			min = a;}
		else
		{
			if(a>max){max=a;}
			if(a<min){min=a;}
		}
		sum = sum+a;
		i = i+1;
	}
	write(max);
	write(min);
	sum = sum/5;
	write(sum);
}
