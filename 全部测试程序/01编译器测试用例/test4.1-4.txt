// 输入a和b，输出最大公约数
// 测试函数的递归调用
int gcd(int, int);

main()
{
    int a;
    int b;
    int c;
    a=read();
    b = read();
    c = gcd(a, b);
    write(c);
    return 0;
}
int gcd(int x, int y)
{
    if(y == 0)
    {
        return x;
    }
    else
    {
        return gcd(y, x-x/y*y);
    }
    return 0;
}
