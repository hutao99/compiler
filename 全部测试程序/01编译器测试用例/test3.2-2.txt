// 输入数字num，判断是不是素数
// 测试函数调用，基本分支，循环结构
int isPrime(int);

main()
{
    int num;
    int res;
    num = read();
    if(isPrime(num) == 1)
    {
        res = 1;
    }
    else
    {
        res = 0;
    }
    write(res);
    return 0;
}
int isPrime(int x)
{
    int i;
    int yes;
    i = 2;
    if(x < 2)
    {
        yes = 0;
    }
    else
    {
        yes = 1;
    }
    while(i * i <= x)
    {
        if(x / i * i == x)
        {
            yes = 0;
        }
        i = i + 1;
    }
    return yes;
}
